Przydatne gówno JavaScript - INF.03

1. Deklaracja zmiennych

W JavaScript można deklarować zmienne za pomocą var, let oraz const. Przykłady:

// Deklaracja zmiennej za pomocą var
var zmienna1 = 'Hello, World!';

// Deklaracja zmiennej za pomocą let
let zmienna2 = 42;

// Deklaracja stałej za pomocą const
const zmienna3 = true;

2. Funkcje

Funkcje w JavaScript mogą być deklarowane na kilka sposobów. Przykłady:

// Deklaracja funkcji
function dodaj(a, b) {
    return a + b;
}

// Funkcja anonimowa przypisana do zmiennej
const odejmij = function(a, b) {
    return a - b;
};

// Funkcja strzałkowa
const pomnoz = (a, b) => a * b;

3. Pętle

JavaScript obsługuje różne rodzaje pętli. Przykłady:

// Pętla for
for (let i = 0; i < 10; i++) {
    console.log(i);
}

// Pętla while
let j = 0;
while (j < 10) {
    console.log(j);
    j++;
}

// Pętla do...while
let k = 0;
do {
    console.log(k);
    k++;
} while (k < 10);

4. Warunki

Instrukcje warunkowe w JavaScript. Przykłady:

// Instrukcja if...else
let x = 10;
if (x > 5) {
    console.log('x jest większe niż 5');
} else {
    console.log('x jest mniejsze lub równe 5');
}

// Instrukcja switch
let y = 'apple';
switch (y) {
    case 'banana':
        console.log('To jest banan');
        break;
    case 'apple':
        console.log('To jest jabłko');
        break;
    default:
        console.log('Nieznany owoc');
}

5. Tablice

Praca z tablicami w JavaScript. Przykłady:

// Tworzenie tablicy
let tablica = [1, 2, 3, 4, 5];

// Dostęp do elementów tablicy
console.log(tablica[0]); // 1

// Iteracja po tablicy
tablica.forEach(element => {
    console.log(element);
});

6. Obiekty

Tworzenie i manipulacja obiektami w JavaScript. Przykłady:

// Tworzenie obiektu
let osoba = {
    imie: 'Jan',
    nazwisko: 'Kowalski',
    wiek: 30
};

// Dostęp do właściwości obiektu
console.log(osoba.imie); // Jan

// Dodawanie nowej właściwości
osoba.email = 'jan.kowalski@example.com';

7. Manipulacja DOM

Manipulacja elementami DOM za pomocą JavaScript. Przykłady:

// Pobieranie elementu
let element = document.getElementById('moj-element');

// Zmiana tekstu elementu
element.textContent = 'Nowy tekst';

// Dodawanie klasy do elementu
element.classList.add('nowa-klasa');

8. Obsługa zdarzeń

Obsługa zdarzeń w JavaScript. Przykłady:

// Dodawanie nasłuchiwacza zdarzeń
let przycisk = document.getElementById('moj-przycisk');
przycisk.addEventListener('click', () => {
    alert('Przycisk został kliknięty!');
});

9. Asynchroniczność

Praca z kodem asynchronicznym w JavaScript. Przykłady:

// Użycie funkcji setTimeout
setTimeout(() => {
    console.log('To jest opóźniona wiadomość');
}, 2000);

// Użycie funkcji fetch do pobierania danych
fetch('https://api.example.com/dane')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error('Błąd:', error));

10. Promisy

Praca z promisami w JavaScript. Przykłady:

// Tworzenie promisa
let obietnica = new Promise((resolve, reject) => {
    let sukces = true;
    if (sukces) {
        resolve('Operacja zakończona sukcesem');
    } else {
        reject('Operacja zakończona niepowodzeniem');
    }
});

// Obsługa promisa
obietnica
    .then(wiadomosc => console.log(wiadomosc))
    .catch(error => console.error(error));

11. Async/Await

Użycie async/await do pracy z kodem asynchronicznym. Przykłady:

// Funkcja asynchroniczna
async function pobierzDane() {
    try {
        let response = await fetch('https://api.example.com/dane');
        let data = await response.json();
        console.log(data);
    } catch (error) {
        console.error('Błąd:', error);
    }
}

// Wywołanie funkcji
pobierzDane();

12. Moduły

Importowanie i eksportowanie modułów w JavaScript. Przykłady:

// Eksportowanie funkcji w module
export function dodaj(a, b) {
    return a + b;
}

// Importowanie funkcji z modułu
import { dodaj } from './modul.js';
console.log(dodaj(2, 3)); // 5

13. AJAX

Użycie AJAX do komunikacji z serwerem. Przykłady:

// Tworzenie obiektu XMLHttpRequest
let xhr = new XMLHttpRequest();
xhr.open('GET', 'https://api.example.com/dane', true);
xhr.onload = function() {
    if (xhr.status === 200) {
        console.log(JSON.parse(xhr.responseText));
    } else {
        console.error('Błąd:', xhr.statusText);
    }
};
xhr.onerror = function() {
    console.error('Błąd sieci');
};
xhr.send();

14. Formularze

Praca z formularzami w JavaScript. Przykłady:

// Pobieranie danych z formularza
let formularz = document.getElementById('moj-formularz');
formularz.addEventListener('submit', function(event) {
    event.preventDefault();
    let dane = new FormData(formularz);
    console.log(dane.get('nazwa-pola'));
});

15. Walidacja

Walidacja danych w JavaScript. Przykłady:

// Prosta walidacja formularza
let emailInput = document.getElementById('email');
emailInput.addEventListener('input', function() {
    if (emailInput.validity.typeMismatch) {
        emailInput.setCustomValidity('Proszę podać poprawny adres email.');
    } else {
        emailInput.setCustomValidity('');
    }
});
